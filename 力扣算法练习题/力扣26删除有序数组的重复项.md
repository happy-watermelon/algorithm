class Solution {
    public int removeDuplicates(int[] nums) {
        int len=nums.length;
        int res=nums[len-1]+1,index=0;
        for(int i=0;i<len;i++){
            if(nums[i]!=res){
                res=nums[i];
                nums[index]=res;
                index++;
            }
        }
        return index;
    }
}
