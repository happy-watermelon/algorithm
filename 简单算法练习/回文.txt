class Solution {
    public boolean isPalindrome(int x) {
        if (x < 0) {
            return false;//负数一定不是回文
        }
        int tmp = x;
        int n = 0;
        while (x > 0) {
            n = n * 10 + x % 10;
            x = x / 10;
        }//while循环将数字倒置
        return tmp == n;//判断倒置后是否和原数相等
    }
}
