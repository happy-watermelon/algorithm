class Solution {
    public String reverseLeftWords(String s, int n) {
        char []c=s.toCharArray();
        int len=c.length;
        reverse(c,0,len-1);
        reverse(c,len-n,len-1);
        reverse(c,0,len-n-1);
        s=String.copyValueOf(c);
        return s;
    }

    public void reverse(char []c,int start,int end){
        for(int i=start,j=end;i<j;i++,j--){
            char temp=c[i];
            c[i]=c[j];
            c[j]=temp;
        }
    }
}
